605    public static function getFullDifff($str_interval, $dt_menor, $dt_maior, $relative=false,$debug=false):float{
606        if(is_string($dt_menor)) $dt_menor = date_create( $dt_menor);
607        if(is_string($dt_maior)) $dt_maior = date_create( $dt_maior);
608
609        $diff = date_diff( $dt_menor, $dt_maior, !$relative);
610        if($debug)
611            Utility::debug($diff);
612
613        switch( $str_interval){
614            case "y":
615                $total = $diff->y + $diff->m / 12 + $diff->d / 365.25;
616                break;
617            case "m":
618                $total= $diff->y * 12 + $diff->m + $diff->d/30 + $diff->h / 24;
619                break;
620            case "d":
621                $total = $diff->y * 365.25 + $diff->m * 30 + $diff->d + $diff->h/24 + $diff->i / 60;
622                break;
623            case "h":
624                $total = ($diff->y * 365.25 + $diff->m * 30 + $diff->d) * 24 + $diff->h + $diff->i/60;
625                break;
626            case "i":
627                $total = (($diff->y * 365.25 + $diff->m * 30 + $diff->d) * 24 + $diff->h) * 60 + $diff->i + $diff->s/60;
628                break;
629            case "s":
630                $total = ((($diff->y * 365.25 + $diff->m * 30 + $diff->d) * 24 + $diff->h) * 60 + $diff->i)*60 + $diff->s;
631                break;
632           }
633        if( $diff->invert)
634            return -1 * $total;
635        else
636            return $total;
637    }
